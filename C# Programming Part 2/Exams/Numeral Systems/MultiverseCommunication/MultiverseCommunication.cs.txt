namespace MultiverseCommunication
{
    using System;

    public class MultiverseCommunication
    {
        private static string[] numeralSystem = new string[13]
        {
            "CHU", "TEL", "OFT", "IVA", "EMY", "VNB", "POQ", "ERI", "CAD", "K-A", "IIA", "YLO", "PLA"
        };
        private const int NumeralSystemType = 13;
        private const int TokenLength = 3;

        public static void Main()
        {
            string input = Console.ReadLine();
            long resultNumber = 0;
            int powIndex = 0;

            for (int i = input.Length - 1; i >= 2; i--, powIndex++)
            {
                string currentToken = input.Substring(i - 2, TokenLength);
                int tokenValue = Array.IndexOf(numeralSystem, currentToken);

                resultNumber += tokenValue * (long)Math.Pow(NumeralSystemType, powIndex);
                i -= 2;
            }

            Console.WriteLine(resultNumber);
        }
    }
}
